-- Neo4j queries executed in this session

CALL db.schema.relTypeProperties() YIELD relType, propertyName, propertyTypes RETURN relType, propertyName, propertyTypes
MATCH (s)-[r]->(t) RETURN DISTINCT labels(s) AS fromLabels, type(r) AS relType, labels(t) AS toLabels
