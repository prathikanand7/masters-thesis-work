// Component: Synchronization

bool SynchronizedRunner::runSynchronized(Aggregator& agg)
{
    auto runnableObjects = agg.getAll<IRunnableObject>();  // → Aggregation

    RunStage lastRunStage = RunStage::SYNCHRONIZE;
    while (lastRunStage != RunStage::FINAL)
    {
        // ... wait for stage signal from master ...
        for (auto it : runnableObjects)
            if (it->run(stage)) return true;

        lastRunStage = stage;
        CPSLogger::instance()->flush();  // → Logging: instance(), flush()
        synchronizer->finishedStage.post();
    }
    return false;
}
