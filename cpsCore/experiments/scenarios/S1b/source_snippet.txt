// Component: Synchronization

bool
SynchronizedRunnerMaster::runStage(RunStage stage)
{
	using namespace boost::interprocess;
	mapped_region region(sync_, read_write);
	auto synchronizer = static_cast<Synchronizer*>(region.get_address());

	std::unique_lock<boost::interprocess::interprocess_mutex> lock(synchronizer->runStageMutex);
	synchronizer->runStage = stage;
	synchronizer->runStageChanged.notify_all();
	lock.unlock();

	boost::posix_time::ptime timeout = boost::get_system_time()
			+ boost::posix_time::milliseconds(
					std::chrono::duration_cast<Milliseconds>(timeout_).count());

	for (int i = 0; i < numOfRunners_; ++i)
	{
		if (!synchronizer->finishedStage.timed_wait(timeout))
		{
			CPSLOG_ERROR << numOfRunners_ - i << " Module(s) timed out at runstage " << (int) stage;
			std::unique_lock<boost::interprocess::interprocess_mutex> lock(
					synchronizer->runStageMutex);
			synchronizer->failure = true;
			synchronizer->runStageChanged.notify_all();
			lock.unlock();
			return true;
		}
	}
	return false;
}

// === src/Synchronization/SynchronizedRunner.cpp ===
// Component: Synchronization

bool SynchronizedRunner::runSynchronized(Aggregator& agg)
{
    using namespace boost::interprocess;
    mapped_region region(sync_, read_write);
    auto synchronizer = static_cast<Synchronizer*>(region.get_address());

    auto runnableObjects = agg.getAll<IRunnableObject>();
    // ^ Synchronization → Aggregation : Aggregator.getAll
    // Called by EACH runner thread — runner1 and runner2 both call this.
    // Fires TWICE total (once per thread).

    RunStage lastRunStage = RunStage::SYNCHRONIZE;
    while (lastRunStage != RunStage::FINAL)
    {
        // ... wait for master to set stage ...
        std::unique_lock<...> lock(synchronizer->runStageMutex);
        if (synchronizer->failure) return true;  // early exit on failure, no flush
        stage = synchronizer->runStage;
        // ... wait on condition variable if stage unchanged ...
        if (synchronizer->failure) return true;
        lock.unlock();

        bool failed = false;
        for (auto it : runnableObjects)
            if (it->run(stage)) failed = true;
        if (failed) return true;              // early exit, no flush

        lastRunStage = stage;
        CPSLogger::instance()->flush();       // Synchronize stdio after stage
        // ^ Synchronization → Logging : CPSLogger.flush
        // Fires once per runner per successfully completed stage.
        // runner2 (fast) completes INIT → flush.
        // runner1 (slow, 2s) also eventually completes INIT → flush.
        // = 2 CPSLogger.flush events total in the GT sequence.
        synchronizer->finishedStage.post();   // signal master this runner is done
    }
    return false;
}

// === src/Synchronization/SynchronizedRunnerMaster.cpp ===
// Component: Synchronization

bool SynchronizedRunnerMaster::runStage(RunStage stage)
{
    using namespace boost::interprocess;
    mapped_region region(sync_, read_write);
    auto synchronizer = static_cast<Synchronizer*>(region.get_address());

    // Signal all runners to execute this stage
    std::unique_lock<...> lock(synchronizer->runStageMutex);
    synchronizer->runStage = stage;
    synchronizer->runStageChanged.notify_all();
    lock.unlock();

    boost::posix_time::ptime timeout = boost::get_system_time()
            + boost::posix_time::milliseconds(
                    std::chrono::duration_cast<Milliseconds>(timeout_).count());

    for (int i = 0; i < numOfRunners_; ++i)
    {
        if (!synchronizer->finishedStage.timed_wait(timeout))
        {
            // Timeout: runner did not complete stage in time
            CPSLOG_ERROR << numOfRunners_ - i << " Module(s) timed out at runstage " << (int)stage;
            // ^ Synchronization → Logging : RAIILogStream.stream
            // Fires when timed_wait expires (runner1 sleeps 2s, timeout is 1s).
            std::unique_lock<...> lock2(synchronizer->runStageMutex);
            synchronizer->failure = true;
            synchronizer->runStageChanged.notify_all();
            lock2.unlock();
            return true;
        }
    }
    return false;
}
